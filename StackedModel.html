# --- Level 1 Meta-Model Training (The Stack) ---

# 1. Load the Partitioned Stacked Data
# These were created in Step 7
stack_train <- readRDS("depression_stack_train.rds")
stack_test  <- readRDS("depression_stack_test.rds")

# Ensure the target is a factor for classification
stack_train$actual <- as.factor(stack_train$actual)
stack_test$actual  <- as.factor(stack_test$actual)

# 2. Define the Financial Cost Matrix
# The matrix format for C5.0 is: [Predicted Class, Actual Class]
# Position [1,2] is Actual "1" Predicted "0" (False Negative)
# Position [2,1] is Actual "0" Predicted "1" (False Positive)

cost_matrix <- matrix(c(0, 800, 251, 0), nrow = 2, byrow = TRUE)

# Explicitly label the rows and columns so R doesn't get confused
rownames(cost_matrix) <- c("0", "1") # Predicted
colnames(cost_matrix) <- c("0", "1") # Actual

# Verify the matrix looks like this:
#      Actual:0  Actual:1
# Pred:0    0      800    <-- Cost of False Negative
# Pred:1   251      0     <-- Cost of False Positive
print(cost_matrix)
##     0   1
## 0   0 800
## 1 251   0
# 3. Build the Meta-Decision Tree (C5.0)
library(C50)
# We use the dot '.' to include all expert probabilities (logreg, knn, ann, svm, tree)
m_meta_tree <- C5.0(actual ~ ., 
                    data = stack_train, 
                    costs = cost_matrix)

# Visualize the 'Expert Logic'
# This is great for Mike to see which threshold triggers a "Yes" diagnosis
plot(m_meta_tree, main = "Meta-Model Logic (Council of Experts)")

# 4. Build the Meta-Random Forest
# RF is excellent for stacking because it handles correlations between experts
library(randomForest)
## randomForest 4.7-1.2
## Type rfNews() to see new features/changes/bug fixes.
set.seed(12345)
m_meta_rf <- randomForest(actual ~ ., 
                          data = stack_train, 
                          ntree = 500, 
                          importance = TRUE)

# 5. Expert Importance Analysis
# This is the "Gold" deliverable: It shows which of your models was actually the best!
varImpPlot(m_meta_rf, main = "Expert Importance in the Depression Stack")

# 6. Save the Meta-Models
saveRDS(m_meta_tree, "depression_meta_model_tree.rds")
saveRDS(m_meta_rf, "depression_meta_model_rf.rds")
